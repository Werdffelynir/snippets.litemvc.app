[h]Access via attributes[/h] [code=css]/* Псевдо типы применяем к ссылкам. */ a:link{} /* Определяет стиль для обычной непосещенной ссылки.*/ a:visited{} /* Определяет стиль для посещенной ссылки.*/ a:active{} /* Определяет стиль для активной ссылки. Активной ссылка становится при нажатии на нее.*/ a:hover{} /* Определяет стиль для ссылки при наведении на нее мышью.*/ p:hover{} /* параграф будет менять визуальны стиль при наведении курсора.*/ /* К всем изображения в которых задан атрибут alt="" */ img[alt] {border:1px solid red;} /* К всем елементам содержащий атребут type с свойством text */ [type="text"] {font-size:18px;} /* К всем елементам с классом img ... не пашит */ [class~="img"] {font-size:22px; color: #FF0033;} /* К всем с началом на http://*/ [href^="http://"] {color:#66CCCC;} /* К всем с концом на .com */ [href$=".com"] {font-size:18px;} /* К всем с title="man" */ [title*="man"] {background-color:#FFFFCC;} /* К всем сылкам в параграфах, в е параграфов не применится */ p>a {color:red;} /* Для выбора элементов, расположенных на одном уровне, имеющие одного родителя */ strong + em {font-style:italic;} /* для imput в фокусе */ input:focus {color: #66FF00; border-color: #000066; } /* К первой букве абзца */ p:first-letter {font-family: Verdana; font-size: 24px; color: #333;} /* Выделяет красным ссылки, которые находятся внутри параграфа.*/ p>a {color:red;} /* Выделяет первый элемент */ .menu li:first /* Выделяет последний элемент */ .menu li:last[/code]
